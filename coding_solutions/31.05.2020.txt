#include<stdio.h>
voidswap(int*xp,int*yp)
{
inttemp=*xp;
*xp=*yp;
*yp=temp;
}
intbubbleSort(intarr[],intn)
{
inti,j,count=0;
intswapped;
for(i=0;i<n-1;i++)
{
swapped=0;
for(j=0;j<n-i-1;j++)
{
if(arr[j]>arr[j+1])
{
swap(&arr[j],&arr[j+1]);
swapped=1;
count++;
}
}
if(swapped==0)
break;
}
returncount;
}
