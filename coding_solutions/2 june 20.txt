Given an array of positive integers. Write a C Program to find inversion count of array.
#include <stdio.h>

int getInvCount(int arr[], int n) 
{ 
    int inv_count = 0; 
    for (int i = 0; i < n - 1; i++) 
        for (int j = i + 1; j < n; j++) 
            if (arr[i] > arr[j]) 
                inv_count++; 
  
    return inv_count; 
} 
  
int main() 
{ 
    int arr[100],i=0,n=0;
    printf("enter the number of elements");
    scanf("%d",&n);
    printf("enter the array");
    for(i=0;i<n;i++)
    {
	scanf("%d",&arr[i]);
    }  
    int count = getInvCount(arr, n); 
    printf(" Number of inversions are %d",count);
    return 0; 
}
